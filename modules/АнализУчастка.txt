Функция ВыполнитьPOSTЗапрос(Путь, ТелоЗапроса = "") Экспорт
	
	Соединение = Новый HTTPСоединение("localhost", 8000);
	Запрос = Новый HTTPЗапрос(Путь);
	Запрос.Заголовки.Вставить("Accept", "application/json");
	Запрос.Заголовки.Вставить("Content-Type", "application/json");
	
	Если НЕ ПустаяСтрока(ТелоЗапроса) Тогда
		Запрос.УстановитьТелоИзСтроки(ТелоЗапроса, КодировкаТекста.UTF8);
	КонецЕсли;
	
	Ответ = Соединение.ОтправитьДляОбработки(Запрос);
	
	Если Ответ.КодСостояния <> 200 И Ответ.КодСостояния <> 201 Тогда
		ВызватьИсключение "Ошибка HTTP: "
			+ Строка(Ответ.КодСостояния)
			+ ". "
			+ Ответ.ПолучитьТелоКакСтроку();
	КонецЕсли;
	
	Возврат Ответ.ПолучитьТелоКакСтроку();

Функция КрасивыйВердикт(КлассМодели) Экспорт
	
	Если ПустаяСтрока(КлассМодели) Тогда
		Возврат "Анализ не выполнен";
	КонецЕсли;
	
	КлассМодели = НРег(СокрЛП(КлассМодели));
	
	Если КлассМодели = "agriculture" Тогда
		Возврат "Категория: сельскохозяйственное использование. На снимке видны признаки обработки или регулярной структуры территории.";
	ИначеЕсли КлассМодели = "overgrown" Тогда
		Возврат "Категория: зарастание. Участок покрыт растительностью без выраженных признаков активного использования.";
	ИначеЕсли КлассМодели = "built_up" Тогда
		Возврат "Категория: застроенная территория. На участке преобладают сооружения или плотная инфраструктура.";
	ИначеЕсли КлассМодели = "bare" Тогда
		Возврат "Категория: открытый грунт. Выраженная растительность и плотная застройка отсутствуют.";
	Иначе
		Возврат "Результат модели: " + КлассМодели;
	КонецЕсли;
